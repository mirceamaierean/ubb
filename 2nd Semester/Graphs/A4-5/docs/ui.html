353        """

363                print("The number of edges is: " +

365            elif command == "2":
366                self.print_graph()
367            elif command == "3":
368                self.check_if_edge_exists()
369            elif command == "4":
370                self.get_in_degree_and_out_degree()
371            elif command == "5":
372                self.parse_outbound_edges()
373            elif command == "6":
374                self.parse_inbound_edges()
375            elif command == "7":
376                self.get_endpoints_of_edge()
377            elif command == "8":
378                self.modify_cost_of_edge()
379            elif command == "9":
380                self.add_edge()
381            elif command == "10":
382                self.remove_edge()
383            elif command == "11":
384                self.add_vertex()
385            elif command == "12":
386                self.remove_vertex()
387            elif command == "13":
388                self.copy_graph()
389            elif command == "14":
390                self.read_graph_from_file()
391            elif command == "15":
392                self.write_graph_to_file()
393            elif command == "16":
394                self.generate_random_graph()
395            elif command == "17":
396                self.print_vertices()
397            elif command == "18":
398                self.print_the_cost()
399            elif command == "19":
400                self.minimum_cost_spanning_tree()
401            elif command == "20":
402                self.print_minimum_cost_walk_in_dag()
403            elif command == "21":
404                self.print_number_of_paths()
405            elif command == "22":
406                self.print_hamiltonian_cycle()
407            elif command == "23":
408                filename = "graph" + \
409                    str(self.__service.graph.get_number_of_vertices()) + "_modif.txt"
410                self.__service.write_graph_to_file(filename)
411                print("Goodbye!")
412                return
413            else:
414                print("Invalid command!. Please try again!")
